let progress = JSON.parse(localStorage.getItem('ecm-progress')||'{}');
let bookmarks = JSON.parse(localStorage.getItem('ecm-bookmarks')||'[]');
let currentModalIdx = -1;

function saveProgress(){ localStorage.setItem('ecm-progress', JSON.stringify(progress)); }
function saveBookmarks(){ localStorage.setItem('ecm-bookmarks', JSON.stringify(bookmarks)); }

function renderProgress(){
  const done = Object.values(progress).filter(Boolean).length;
  const pct = Math.round(done/TOPICS.length*100);
  const fill = document.getElementById('prog-fill');
  const pctEl = document.getElementById('prog-pct');
  if(fill) fill.style.width = pct+'%';
  if(pctEl) pctEl.textContent = done+' / '+TOPICS.length;
  const dots = document.getElementById('prog-dots');
  if(!dots) return;
  dots.innerHTML = '';
  TOPICS.forEach((_,i)=>{
    const d = document.createElement('div');
    d.style.cssText = `height:4px;border-radius:2px;cursor:pointer;transition:background .2s;background:${progress[i]?'var(--blue,#1a5fa8)':'var(--bd,#e2e8f0)'};`;
    d.title = TOPICS[i].title;
    d.onclick = ()=>{ progress[i]=!progress[i]; saveProgress(); renderProgress(); };
    dots.appendChild(d);
  });
}

function renderBookmarks(){
  const body = document.getElementById('bm-body');
  const empty = document.getElementById('bm-empty');
  if(!body) return;
  const items = body.querySelectorAll('.bm-item');
  items.forEach(el=>el.remove());
  if(!bookmarks.length){ if(empty) empty.style.display='block'; return; }
  if(empty) empty.style.display='none';
  [...bookmarks].reverse().forEach(idx=>{
    const t = TOPICS[idx]; if(!t) return;
    const el = document.createElement('div');
    el.className = 'bm-item';
    el.style.cssText = 'display:flex;align-items:center;justify-content:space-between;padding:6px 0;border-bottom:1px solid var(--bd,#e2e8f0);cursor:pointer;';
    el.innerHTML = `<span style="font-size:12px;color:var(--text,#1e293b);transition:color .13s;" onmouseover="this.style.color='var(--blue)'" onmouseout="this.style.color='var(--text,#1e293b)'">${t.num} ${t.title}</span><button onclick="removeBookmark(${idx})" style="background:none;border:none;color:var(--text3,#94a3b8);cursor:pointer;font-size:14px;padding:0 4px;">✕</button>`;
    el.querySelector('span').onclick = ()=>openModal(idx);
    body.insertBefore(el, body.children[1]||null);
  });
}

function toggleBookmark(){
  if(currentModalIdx<0) return;
  const idx = bookmarks.indexOf(currentModalIdx);
  if(idx>=0) bookmarks.splice(idx,1); else bookmarks.push(currentModalIdx);
  saveBookmarks(); renderBookmarks(); updateBmBtn();
}
function removeBookmark(idx){ bookmarks=bookmarks.filter(i=>i!==idx); saveBookmarks(); renderBookmarks(); }
function updateBmBtn(){
  const btn = document.getElementById('m-bm-btn');
  if(!btn) return;
  const saved = bookmarks.includes(currentModalIdx);
  btn.textContent = saved ? '★ 저장됨' : '☆ 북마크';
  btn.style.background = saved ? 'rgba(217,119,6,.2)' : 'none';
  btn.style.borderColor = saved ? 'rgba(217,119,6,.6)' : 'rgba(255,255,255,.4)';
  btn.style.color = saved ? '#fbbf24' : 'rgba(255,255,255,.85)';
}

// ══ CALCULATOR ══
function calcRC(){
  if(!document.getElementById('c-r1')) return;
  const R1=parseFloat(document.getElementById('c-r1').value)/1000;
  const C1=parseFloat(document.getElementById('c-c1').value);
  const R0=parseFloat(document.getElementById('c-r0').value)/1000;
  const di=parseFloat(document.getElementById('c-di').value);
  if(isNaN(R1)||isNaN(C1)||isNaN(R0)||isNaN(di)) return;
  const tau=R1*C1;
  document.getElementById('r-tau').textContent=tau.toFixed(1)+' s';
  document.getElementById('r-4tau').textContent=(4*tau).toFixed(1)+' s (약 '+(Math.round(4*tau/60))+'분)';
  document.getElementById('r-dv0').textContent=(R0*di*1000).toFixed(1)+' mV';
  document.getElementById('r-dvinf').textContent=((R0+R1)*di*1000).toFixed(1)+' mV';
  document.getElementById('r-f1').textContent=Math.exp(-1/tau).toFixed(6);
}
calcRC();

// ══ SIMULATOR (Chart.js) ══
let simChart=null;
function runSim(){
  const I=parseFloat(document.getElementById('s-I').value)||20;
  const R0=(parseFloat(document.getElementById('s-R0').value)||8.2)/1000;
  const R1=(parseFloat(document.getElementById('s-R1').value)||15.8)/1000;
  const C1=parseFloat(document.getElementById('s-C1').value)||38000;
  const ocv=4.1;
  const labels=[],dataV=[],dataOCV=[];
  let irc=0;
  for(let t=0;t<=80;t++){
    labels.push(t+'s');
    dataOCV.push(ocv);
    if(t<5){ dataV.push(ocv); irc=0; }
    else{
      const F=Math.exp(-1/(R1*C1));
      irc=F*irc+(1-F)*I;
      dataV.push(Number((ocv-R1*irc-R0*I).toFixed(4)));
    }
  }
  const ctx=document.getElementById('simChart');
  if(!ctx) return;
  if(simChart) simChart.destroy();
  const isDark = document.documentElement.getAttribute('data-theme')==='dark';
  const gridColor = isDark ? 'rgba(100,116,139,0.2)' : 'rgba(226,232,240,0.8)';
  const textColor = isDark ? '#94a3b8' : '#475569';
  const titleColor = isDark ? '#94a3b8' : '#475569';
  simChart=new Chart(ctx.getContext('2d'),{
    type:'line',
    data:{labels,datasets:[
      {label:'단자 전압 V(t)',data:dataV,borderColor:'#1a5fa8',backgroundColor:'rgba(26,95,168,.07)',borderWidth:2.5,pointRadius:0,fill:true,tension:.15},
      {label:'OCV 기준선',data:dataOCV,borderColor:'#dc2626',borderWidth:1.5,pointRadius:0,fill:false,borderDash:[6,4]},
    ]},
    options:{responsive:true,animation:{duration:200},interaction:{mode:'index',intersect:false},
      plugins:{
        legend:{labels:{font:{size:11},color:textColor}},
        title:{display:true,text:'V(t) 응답 | I='+I+'A, R₀='+(R0*1000).toFixed(1)+'mΩ, τ='+(R1*C1).toFixed(0)+'s',font:{size:11},color:titleColor}
      },
      scales:{
        y:{min:Math.max(3.4,ocv-I*(R0+R1)*1.6),max:ocv+0.06,
          title:{display:true,text:'Voltage (V)',font:{size:11},color:textColor},
          ticks:{color:textColor,font:{size:10}},
          grid:{color:gridColor}},
        x:{title:{display:true,text:'Time (s)',font:{size:11},color:textColor},
          ticks:{color:textColor,font:{size:10},maxTicksLimit:10},
          grid:{color:gridColor}}
      }
    }
  });
}
// Chart.js ready guard
function initSimulator(){
  if(typeof Chart !== 'undefined'){
    runSim();
  } else {
    setTimeout(initSimulator, 100);
  }
}
initSimulator();

// ══ WEATHER ══
const WMO={0:'맑음',1:'대체로 맑음',2:'구름 조금',3:'흐림',45:'안개',51:'이슬비',61:'비',63:'비',71:'눈',80:'소나기',95:'뇌우'};
const WIC={0:'☀️',1:'🌤',2:'⛅',3:'☁️',45:'🌫',51:'🌦',61:'🌧',63:'🌧',71:'❄️',80:'🌦',95:'⛈'};
async function loadWeather(){
  try{
    const r=await fetch('https://api.open-meteo.com/v1/forecast?latitude=35.98&longitude=126.71&current=temperature_2m,weathercode,wind_speed_10m,relative_humidity_2m&timezone=Asia%2FSeoul&forecast_days=1');
    const d=await r.json(); const c=d.current;
    const code=c.weathercode;
    const tmp=Math.round(c.temperature_2m);
    const wind=Math.round(c.wind_speed_10m);
    const hum=Math.round(c.relative_humidity_2m);
    const el=id=>document.getElementById(id);
    if(el('w-temp')) el('w-temp').textContent=tmp+'°C';
    if(el('w-icon')) el('w-icon').textContent=WIC[code]||'🌡';
    if(el('w-desc')) el('w-desc').textContent=WMO[code]||'날씨';
    if(el('w-wind')) el('w-wind').textContent=wind;
    if(el('w-humid')) el('w-humid').textContent=hum;
    if(el('wt-top')) el('wt-top').textContent=(WIC[code]||'🌡')+' 군산 '+tmp+'°C · '+(WMO[code]||'');
  }catch(e){ const el=document.getElementById('w-desc'); if(el) el.textContent='불러오기 실패'; }
}
loadWeather(); setInterval(loadWeather,600000);
